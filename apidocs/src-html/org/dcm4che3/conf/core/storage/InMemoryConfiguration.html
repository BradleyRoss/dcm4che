041package org.dcm4che3.conf.core.storage;
042
043import org.dcm4che3.conf.core.api.Configuration;
044import org.dcm4che3.conf.core.api.ConfigurationException;
045import org.dcm4che3.conf.core.Nodes;
046import org.dcm4che3.conf.core.api.Path;
047
048import java.util.HashMap;
049import java.util.Iterator;
050import java.util.Map;
051
052/**
053 * @author Roman K
054 */
055public class InMemoryConfiguration implements Configuration {
056
057    private final Map<String, Object> root;
058
059    public InMemoryConfiguration() {
060        this.root = new HashMap<String, Object>();
061    }
062
063    public InMemoryConfiguration(Map<String, Object> root) {
064        this.root = root;
065    }
066
067    
068    public Map<String, Object> getConfigurationRoot() throws ConfigurationException {
069        return root;
070    }
071
072    
073    public Object getConfigurationNode(Path path, Class configurableClass) throws ConfigurationException {
074        return Nodes.deepCloneNode(Nodes.getNode(root, path.getPathItems()));
075    }
076
077    
078    public boolean nodeExists(Path path) throws ConfigurationException {
079        return Nodes.nodeExists(root, path.getPathItems());
080    }
081
082    
083    public void refreshNode(Path path) throws ConfigurationException {
084    }
085
086    
087    public void persistNode(Path path, Map<String, Object> configNode, Class configurableClass) throws ConfigurationException {
088        if (!Path.ROOT.equals(path))
089            Nodes.replaceNode(getConfigurationRoot(), (Map<String, Object>) Nodes.deepCloneNode(configNode), path.getPathItems());
090        else {
091            root.clear();
092            root.putAll(configNode);
093        }
094    }
095
096    
097    public void removeNode(Path path) throws ConfigurationException {
098        Nodes.removeNode(getConfigurationRoot(), path.getPathItems());
099    }
100
101    
102    public Path getPathByUUID(String uuid) {
103        throw new ConfigurationException("Unexpected error - uuid index is missing");
104    }
105
106    
107    public Iterator search(String liteXPathExpression) throws IllegalArgumentException, ConfigurationException {
108        final Iterator search = Nodes.search(root, liteXPathExpression);
109        return new Iterator() {
110            
111            public boolean hasNext() {
112                return search.hasNext();
113            }
114
115            
116            public Object next() {
117                return Nodes.deepCloneNode(search.next());
118            }
119
120            
121            public void remove() {
122                // noop
123            }
124        };
125    }
126
127    
128    public void lock() {
129    }
130
131    
132    public void runBatch(Batch batch) {
133        batch.run();
134    }
135}
