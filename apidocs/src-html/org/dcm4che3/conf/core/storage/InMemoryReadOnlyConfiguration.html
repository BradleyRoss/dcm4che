041package org.dcm4che3.conf.core.storage;
042
043import org.dcm4che3.conf.core.api.Configuration;
044import org.dcm4che3.conf.core.api.ConfigurationException;
045import org.dcm4che3.conf.core.Nodes;
046import org.dcm4che3.conf.core.api.Path;
047
048import java.util.Iterator;
049import java.util.Map;
050
051/**
052 * @author Roman K
053 */
054public class InMemoryReadOnlyConfiguration implements Configuration {
055
056    private final Map<String, Object> root;
057
058    public InMemoryReadOnlyConfiguration(Map<String, Object> root) {
059        this.root = root;
060    }
061
062    @Override
063    public Map<String, Object> getConfigurationRoot() throws ConfigurationException {
064        return root;
065    }
066
067    @Override
068    public Object getConfigurationNode(Path path, Class configurableClass) throws ConfigurationException {
069        return Nodes.getNode(root, path.getPathItems());
070    }
071
072    @Override
073    public boolean nodeExists(Path path) throws ConfigurationException {
074        return Nodes.nodeExists(root, path.getPathItems());
075    }
076
077    @Override
078    public void refreshNode(Path path) throws ConfigurationException {
079    }
080
081    @Override
082    public void persistNode(Path path, Map<String, Object> configNode, Class configurableClass) throws ConfigurationException {
083        throw new RuntimeException("Configuration is read-only");
084    }
085
086    @Override
087    public void removeNode(Path path) throws ConfigurationException {
088        throw new RuntimeException("Configuration is read-only");
089    }
090
091    @Override
092    public Path getPathByUUID(String uuid) {
093        throw new ConfigurationException("Unexpected error - uuid index is missing");
094    }
095
096    @Override
097    public Iterator search(String liteXPathExpression) throws IllegalArgumentException, ConfigurationException {
098        return Nodes.search(root, liteXPathExpression);
099    }
100
101    @Override
102    public void lock() {
103    }
104
105    @Override
106    public void runBatch(Batch batch) {
107        batch.run();
108    }
109}
