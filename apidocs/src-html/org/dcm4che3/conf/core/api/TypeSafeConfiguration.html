041package org.dcm4che3.conf.core.api;
042
043import org.dcm4che3.conf.core.context.ContextFactory;
044import org.dcm4che3.conf.core.context.LoadingContext;
045import org.dcm4che3.conf.core.api.internal.BeanVitalizer;
046
047import javax.naming.ldap.ControlFactory;
048
049/**
050 * This API is work in progress, do not use it just yet.
051 *
052 * Highest level core configuration API.
053 *
054 * @param <R> Configuration root class
055 */
056public interface TypeSafeConfiguration<R> {
057
058    <T> T load(Path path, Class<T> clazz);
059    <T> T load(Path path, Class<T> clazz, LoadingContext ctx);
060
061    <T> void save(Path path, T object, Class<T> clazz);
062
063    /**
064     * @param uuid uuid of the configurable object
065     * @param clazz expected class of the configured object
066     * @return configured object or null if an object with this UUID not found
067     */
068    <T> T findByUUID(String uuid, Class<T> clazz);
069
070    /**
071     * @param uuid uuid of the configurable object
072     * @param clazz expected class of the configured object
073     * @return configured object or null if an object with this UUID not found
074     */
075    <T> T findByUUID(String uuid, Class<T> clazz, LoadingContext ctx);
076
077    Configuration getLowLevelAccess();
078    BeanVitalizer getVitalizer();
079    ContextFactory getContextFactory();
080
081    Class<R> getRootClass();
082}
