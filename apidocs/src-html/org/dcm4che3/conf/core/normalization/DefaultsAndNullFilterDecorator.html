043
044import org.dcm4che3.conf.core.DelegatingConfiguration;
045import org.dcm4che3.conf.core.adapters.ArrayTypeAdapter;
046import org.dcm4che3.conf.core.api.ConfigurableProperty;
047import org.dcm4che3.conf.core.api.Configuration;
048import org.dcm4che3.conf.core.api.ConfigurationException;
049import org.dcm4che3.conf.core.api.Path;
050import org.dcm4che3.conf.core.api.internal.ConfigProperty;
051import org.dcm4che3.conf.core.api.internal.BeanVitalizer;
052import org.dcm4che3.conf.core.api.internal.ConfigReflection;
053import org.dcm4che3.conf.core.context.ContextFactory;
054import org.dcm4che3.conf.core.util.ConfigNodeTraverser;
055import org.dcm4che3.conf.core.util.ConfigNodeTraverser.ConfigNodeTypesafeFilter;
056import org.slf4j.Logger;
057import org.slf4j.LoggerFactory;

059/**
060 *
061 */
062@SuppressWarnings("unchecked")
063public class DefaultsAndNullFilterDecorator extends DelegatingConfiguration {
064
065    public static final Logger log = LoggerFactory.getLogger(DefaultsAndNullFilterDecorator.class);
066
067
068    private final List<Class> allExtensionClasses;
069    private final BeanVitalizer vitalizer;
070    private final ContextFactory contextFactory;
071
072    public DefaultsAndNullFilterDecorator(Configuration delegate, List<Class> allExtensionClasses, BeanVitalizer vitalizer) {
073        super(delegate);
074        this.allExtensionClasses = allExtensionClasses;
075        this.vitalizer = vitalizer;
076        contextFactory = new ContextFactory(vitalizer);
077    }
078
079    @Override
080    public void persistNode(Path path, final Map<String, Object> configNode, final Class configurableClass) throws ConfigurationException {
081
082        ConfigNodeTypesafeFilter filterDefaults = new ConfigNodeTypesafeFilter() {
083            
084            public boolean beforeNode(Map<String, Object> containerNode, Class containerNodeClass, ConfigProperty property) throws ConfigurationException {
085
086                boolean doDelete = false;
087
088                String name = property.getAnnotatedName();
089                if (property.isOlockHash())
090                    name = Configuration.OLOCK_HASH_KEY;
091                if (property.isUuid())
092                    name = Configuration.CONF_STORAGE_SYSTEM_PROP;
093
094                Object value = containerNode.get(name);
095
096                // if the value for a property equals to default, filter it out
097                if (property.getDefaultValue().equals(String.valueOf(value))
098                        || value == null) {
099                    doDelete = true;
100                }
101                // if that is an empty extension map or map
102                else if ((property.isExtensionsProperty()
103                        || property.isMap())
104                        && ((Map) value).size() == 0) {
105                    doDelete = true;
106                }
107                // if that is an empty collection
108                else if ((property.isCollection() && ((Collection) value).size() == 0)) {
109                    doDelete = true;
110
111                } // if that is an array
112                else if (property.isArray()) {
113
114                    // it equals the default value, then filter it out
115                    Object defaultNode = getDefaultValueFromClass(containerNodeClass, property);
116                    if (value.equals(defaultNode)) {
117                        doDelete = true;
118                    }
119                    // if the default is null (i.e. no default) but the array is empty, then also remove it
120                    else if (defaultNode == null && ((Collection) value).size() == 0) {
121                        doDelete = true;
122                    }
123                }
124
125                if (doDelete)
126                    containerNode.remove(name);
127
128                return doDelete;
129            }
130        };
131
132        if (configurableClass != null) {
133            // filter out defaults
134            ConfigNodeTraverser.traverseNodeTypesafe(configNode, ConfigReflection.getDummyPropertyForClass(configurableClass), allExtensionClasses, filterDefaults);
135
136            // generate missing UUIDs
137            ConfigNodeTraverser.traverseNodeTypesafe(configNode, ConfigReflection.getDummyPropertyForClass(configurableClass), allExtensionClasses, new ConfigNodeTypesafeFilter() {
138                
139                public boolean beforeNode(Map<String, Object> containerNode, Class containerNodeClass, ConfigProperty property) throws ConfigurationException {
140                    if (property.isUuid() && (containerNode.get(Configuration.UUID_KEY) == null || "".equals(containerNode.get(Configuration.UUID_KEY)))) {
141                        String newUUID = UUID.randomUUID().toString();
142                        log.warn("Adding a missing UUID [" + newUUID + "] to a " + containerNodeClass.getSimpleName());
143                        containerNode.put(Configuration.UUID_KEY, newUUID);
144                    }
145                    return false;
146                }
147            });
148
149
150        }
151
152        super.persistNode(path, configNode, configurableClass);
153    }
154
155
156    @Override
157    public Object getConfigurationNode(Path path, Class configurableClass) throws ConfigurationException {
158
159        ConfigNodeTypesafeFilter applyDefaults = new ConfigNodeTypesafeFilter() {
160          
161            public boolean beforeNode(Map<String, Object> containerNode, Class containerNodeClass, ConfigProperty property) throws ConfigurationException {
162
163                // if no value for this property, see if there is default and set it
164                if (!containerNode.containsKey(property.getAnnotatedName())) {
165                    String defaultValue = property.getDefaultValue();
166                    if (!defaultValue.equals(ConfigurableProperty.NO_DEFAULT_VALUE)) {
167                        Object normalized = vitalizer.lookupDefaultTypeAdapter(property.getRawClass()).normalize(defaultValue, property, contextFactory.newProcessingContext());
168                        containerNode.put(property.getAnnotatedName(), normalized);
169                        return true;
170                    }
171                    // for null map & extension map - create empty obj
172                    else if ((property.isExtensionsProperty() || property.isMap())) {
173                        containerNode.put(property.getAnnotatedName(), new TreeMap());
174                        return true;
175                    }
176                    // for null collections - create empty list
177                    else if ((property.isCollection())) {
178                        containerNode.put(property.getAnnotatedName(), new ArrayList());
179                        return true;
180                    }
181                    // for null array - set to default value
182                    else if (property.isArray()) {
183                        Object defaultNode = getDefaultValueFromClass(containerNodeClass, property);
184                        containerNode.put(property.getAnnotatedName(), defaultNode);
185                        return true;
186                    }
187                }
188
189                return false;
190            }
191        };
192
193        // fill in default values for properties that are null and have defaults
194        Map<String, Object> node = (Map<String, Object>) super.getConfigurationNode(path, configurableClass);
195        if (configurableClass != null && node != null)
196            ConfigNodeTraverser.traverseNodeTypesafe(node, ConfigReflection.getDummyPropertyForClass(configurableClass), allExtensionClasses, applyDefaults);
197        return node;
198    }
199
200    private Object getDefaultValueFromClass(Class containerNodeClass, ConfigProperty property) {
201        Object defaultValueFromClass;
202        defaultValueFromClass = ConfigReflection.getProperty(vitalizer.newInstance(containerNodeClass), property);
203        return new ArrayTypeAdapter().toConfigNode(defaultValueFromClass, property, contextFactory.newSavingContext());
204    }
205
206
207    @Override
208    public Iterator search(String liteXPathExpression) throws IllegalArgumentException, ConfigurationException {
209        return super.search(liteXPathExpression);
210    }
211}
