040package org.dcm4che3.conf.core.adapters;
041
042import org.dcm4che3.conf.core.api.Configuration;
043import org.dcm4che3.conf.core.api.internal.ConfigProperty;
044import org.dcm4che3.conf.core.context.LoadingContext;
045import org.dcm4che3.conf.core.context.ProcessingContext;
046import org.dcm4che3.conf.core.context.SavingContext;
047import org.dcm4che3.conf.core.api.internal.ConfigTypeAdapter;
048import org.dcm4che3.conf.core.api.ConfigurationException;
049
050import java.util.HashMap;
051import java.util.Map;
052import java.util.Map.Entry;
053import java.util.TreeMap;
054
055@SuppressWarnings("unchecked")
056public class MapTypeAdapter<K, V> implements ConfigTypeAdapter<Map<K, V>, Map<String, Object>> {
057
058
059    public Map<K, V> fromConfigNode(Map<String, Object> configNode, ConfigProperty property, LoadingContext ctx, Object parent) throws ConfigurationException {
060
061        ConfigProperty keyPseudoProperty = property.getPseudoPropertyForGenericsParamater(0);
062        ConfigTypeAdapter<K, String> keyAdapter = (ConfigTypeAdapter<K, String>) ctx.getVitalizer().lookupTypeAdapter(keyPseudoProperty);
063
064        ConfigTypeAdapter<V, Object> valueAdapter;
065        ConfigProperty valuePseudoProperty = property.getPseudoPropertyForGenericsParamater(1);
066        if (property.isCollectionOfReferences())
067            valueAdapter = ctx.getVitalizer().getReferenceTypeAdapter();
068        else
069            valueAdapter = (ConfigTypeAdapter<V, Object>) ctx.getVitalizer().lookupTypeAdapter(valuePseudoProperty);
070
071        Map<K, V> map = new TreeMap<K, V>();
072
073        for (Entry<String, Object> e : configNode.entrySet()) {
074            map.put(keyAdapter.fromConfigNode(keyAdapter.normalize(e.getKey(), keyPseudoProperty, ctx), keyPseudoProperty, ctx, null),
075                    valueAdapter.fromConfigNode(valueAdapter.normalize(e.getValue(),valuePseudoProperty, ctx), valuePseudoProperty, ctx, parent));
076
077        }
078
079        return map;
080    }
081
082    public Map<String, Object> toConfigNode(Map<K, V> object, ConfigProperty property, SavingContext ctx) throws ConfigurationException {
083
084        ConfigProperty keyPseudoProperty = property.getPseudoPropertyForGenericsParamater(0);
085        ConfigTypeAdapter<K, String> keyAdapter = (ConfigTypeAdapter<K, String>) ctx.getVitalizer().lookupTypeAdapter(keyPseudoProperty);
086
087        ConfigTypeAdapter<V, Object> valueAdapter;
088        ConfigProperty valuePseudoProperty = property.getPseudoPropertyForGenericsParamater(1);
089        if (property.isCollectionOfReferences())
090            valueAdapter = ctx.getVitalizer().getReferenceTypeAdapter();
091        else
092            valueAdapter = (ConfigTypeAdapter<V, Object>) ctx.getVitalizer().lookupTypeAdapter(valuePseudoProperty);
093
094        Map<String, Object> configNode = Configuration.NodeFactory.emptyNode();
095
096        for (Entry<K, V> e : object.entrySet()) {
097            configNode.put(
098                    // always forces toString for keys - so Integers, Booleans, etc will get stringified
099                    String.valueOf(keyAdapter.toConfigNode(e.getKey(), keyPseudoProperty, ctx)),
100                    valueAdapter.toConfigNode(e.getValue(), valuePseudoProperty, ctx));
101        }
102
103        return configNode;
104    }
105
106    public Map<String, Object> getSchema(ConfigProperty property, ProcessingContext ctx) throws ConfigurationException {
107
108        Map<String, Object> metadata = new HashMap<String, Object>();
109        Map<String, Object> keyMetadata = new HashMap<String, Object>();
110        Map<String, Object> valueMetadata = new HashMap<String, Object>();
111        Map<String, Object> valueMetadataWrapper = new HashMap<String, Object>();
112
113        metadata.put("type", "object");
114        metadata.put("class", "Map");
115
116        // get adapters
117        ConfigProperty keyPseudoProperty = property.getPseudoPropertyForGenericsParamater(0);
118        ConfigTypeAdapter<K, String> keyAdapter = (ConfigTypeAdapter<K, String>) ctx.getVitalizer().lookupTypeAdapter(keyPseudoProperty);
119
120        ConfigProperty valuePseudoProperty = property.getPseudoPropertyForGenericsParamater(1);
121        ConfigTypeAdapter<V, Object> valueAdapter;
122        if (property.isCollectionOfReferences())
123            valueAdapter = ctx.getVitalizer().getReferenceTypeAdapter();
124        else
125            valueAdapter = (ConfigTypeAdapter<V, Object>) ctx.getVitalizer().lookupTypeAdapter(valuePseudoProperty);
126
127        // fill in key and value metadata
128        keyMetadata.putAll(keyAdapter.getSchema(keyPseudoProperty, ctx));
129        metadata.put("mapkey", keyMetadata);
130
131        valueMetadata.putAll(valueAdapter.getSchema(valuePseudoProperty, ctx));
132        valueMetadataWrapper.put("*", valueMetadata);
133        metadata.put("properties", valueMetadataWrapper);
134
135        return metadata;
136    }
137
138    public Map<String, Object> normalize(Object configNode, ConfigProperty property, ProcessingContext ctx) throws ConfigurationException {
139        if (configNode == null) return new HashMap<String, Object>();
140        return (Map<String, Object>) configNode;
141    }
142}
