041package org.dcm4che3.conf.core.adapters;
042
043import org.dcm4che3.conf.core.api.*;
044import org.dcm4che3.conf.core.context.LoadingContext;
045import org.dcm4che3.conf.core.context.ProcessingContext;
046import org.dcm4che3.conf.core.context.SavingContext;
047import org.dcm4che3.conf.core.api.internal.*;
048import org.dcm4che3.conf.core.util.Extensions;
049import org.slf4j.Logger;
050import org.slf4j.LoggerFactory;
051
052import java.util.HashMap;
053import java.util.List;
054import java.util.Map;
055
056/**
057 * @author Roman K
058 */
059@SuppressWarnings("unchecked")
060public class ExtensionTypeAdapter implements ConfigTypeAdapter<Map<Class<?>, Object>, Map<String, Object>> {
061
062    public static final Logger log = LoggerFactory.getLogger(ExtensionTypeAdapter.class);
063
064    @Override
065    public Map<Class<?>, Object> fromConfigNode(Map<String, Object> configNode, ConfigProperty property, LoadingContext ctx, Object parent) throws ConfigurationException {
066
067        // figure out base extension class
068        Class extensionBaseClass;
069        try {
070            extensionBaseClass = (Class) property.getTypeForGenericsParameter(1);
071        } catch (ClassCastException e) {
072            throw new ConfigurationException("Malformed extensions field, generic parameter 1 must be an extension class (" + property + ")", e);
073        }
074
075        Map<Class<?>, Object> extensionsMap = new HashMap<Class<?>, Object>();
076
077        for (Map.Entry<String, Object> entry : configNode.entrySet()) {
078            try {
079
080                // figure out current extension class
081                List<Class> extensionClasses = ctx.getVitalizer().getExtensionClassesByBaseClass(extensionBaseClass);
082                Class extensionClass = Extensions.getExtensionClassBySimpleName(entry.getKey(), extensionClasses);
083
084                // create empty extension bean
085                Object extension = ctx.getVitalizer().newInstance(extensionClass);
086
087                // proceed with deserialization
088                new ReflectiveAdapter(extension).fromConfigNode((Map<String, Object>) entry.getValue(), ConfigReflection.getDummyPropertyForClass(extensionClass), ctx, parent);
089
090                extensionsMap.put(extensionClass, extension);
091
092            } catch (ClassNotFoundException e) {
093                // noop
094                log.debug("Extension class {} not found", entry.getKey());
095            }
096        }
097
098        return extensionsMap;
099    }
100
101    @Override
102    public Map<String, Object> toConfigNode(Map<Class<?>, Object> object, ConfigProperty property, SavingContext ctx) throws ConfigurationException {
103        Map<String, Object> extensionsMapNode = Configuration.NodeFactory.emptyNode();
104
105        for (Map.Entry<Class<?>, Object> classObjectEntry : object.entrySet()) {
106            Object extensionNode = ctx.getVitalizer().createConfigNodeFromInstance(classObjectEntry.getValue(), classObjectEntry.getKey());
107            extensionsMapNode.put(classObjectEntry.getKey().getSimpleName(), extensionNode);
108        }
109
110        return extensionsMapNode;
111    }
112
113    @Override
114    public Map<String, Object> getSchema(ConfigProperty property, ProcessingContext ctx) throws ConfigurationException {
115        Map<String, Object> metadata = new HashMap<String, Object>();
116        metadata.put("type", "extensionMap");
117        return metadata;
118    }
119
120    @Override
121    public Map<String, Object> normalize(Object configNode, ConfigProperty property, ProcessingContext ctx) throws ConfigurationException {
122        if (configNode == null)
123            return new HashMap<String, Object>();
124        return (Map<String, Object>) configNode;
125    }
126}
