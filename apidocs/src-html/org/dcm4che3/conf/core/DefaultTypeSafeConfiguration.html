041package org.dcm4che3.conf.core;
042
043import org.dcm4che3.conf.core.adapters.DefaultConfigTypeAdapters;
044import org.dcm4che3.conf.core.api.Configuration;
045import org.dcm4che3.conf.core.api.Path;
046import org.dcm4che3.conf.core.api.TypeSafeConfiguration;
047import org.dcm4che3.conf.core.api.internal.BeanVitalizer;
048import org.dcm4che3.conf.core.api.internal.ConfigReflection;
049import org.dcm4che3.conf.core.api.internal.ConfigTypeAdapter;
050import org.dcm4che3.conf.core.context.ContextFactory;
051import org.dcm4che3.conf.core.context.LoadingContext;
052import org.dcm4che3.conf.core.context.Referable;
053import org.dcm4che3.conf.core.util.PathPattern;
054
055import java.util.List;
056import java.util.Map;
057import java.util.concurrent.Future;
058
059/**
060 * @author Roman K
061 */
062@SuppressWarnings("unchecked")
063public class DefaultTypeSafeConfiguration<R> implements TypeSafeConfiguration<R> {
064
065    private static PathPattern referencePattern = new PathPattern(Configuration.REFERENCE_BY_UUID_PATTERN);
066    private final Configuration confStorage;
067    private final Class<R> rootClass;
068    private final BeanVitalizer vitalizer;
069    private ContextFactory contextFactory;
070    private Map<Class, ConfigTypeAdapter> customAdapters;
071
072    public DefaultTypeSafeConfiguration(Configuration configurationStorage, Class<R> rootClass, Map<Class, List<Class>> extensionsByClass) {
073        this.confStorage = configurationStorage;
074        this.rootClass = rootClass;
075        contextFactory = new ContextFactory(this);
076        this.vitalizer = new DefaultBeanVitalizer(extensionsByClass, contextFactory);
077
078    }
079
080    @Override
081    public <T> T load(Path path, Class<T> clazz) {
082        return load(path, clazz, contextFactory.newLoadingContext());
083    }
084
085    @Override
086    public <T> T load(Path path, Class<T> clazz, LoadingContext ctx) {
087
088        Object configurationNode = confStorage.getConfigurationNode(path, clazz);
089
090        if (configurationNode == null) {
091            return null;
092        }
093
094        return vitalizer.newConfiguredInstance((Map<String, Object>) configurationNode, clazz, ctx);
095    }
096
097    public <T> T findByUUID(String uuid, Class<T> clazz) {
098        return findByUUID(uuid, clazz, contextFactory.newLoadingContext());
099    }
100
101    @Override
102    public <T> T findByUUID(String uuid, Class<T> clazz, LoadingContext ctx) {
103
104        // TODO: proper cast checks!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
105
106        // either get existing from the context
107        Referable referable = ctx.getReferable(uuid);
108        if (referable != null) {
109
110            // TODO here
111            return (T) referable.getConfObject();
112        }
113
114        //// or go ahead and load
115
116        Path pathByUUID = confStorage.getPathByUUID(uuid);
117
118        if (pathByUUID == null) return null;
119
120        Map<String, Object> referencedNode = (Map<String, Object>) confStorage.getConfigurationNode(
121                pathByUUID,
122                clazz
123        );
124
125        // TODO here
126        return (T) vitalizer.lookupTypeAdapter(ConfigReflection.getDummyPropertyForClass(clazz))
127                .fromConfigNode(referencedNode, ConfigReflection.getDummyPropertyForClass(clazz), ctx, null);
128
129    }
130
131    @Override
132    public <T> void save(Path path, T object, Class<T> clazz) {
133        Map<String, Object> node = vitalizer.createConfigNodeFromInstance(object, clazz);
134        confStorage.persistNode(path, node, clazz);
135    }
136
137    @Override
138    public Configuration getLowLevelAccess() {
139        return confStorage;
140    }
141
142    @Override
143    public BeanVitalizer getVitalizer() {
144        return vitalizer;
145    }
146
147    @Override
148    public ContextFactory getContextFactory() {
149        return contextFactory;
150    }
151
152    @Override
153    public Class<R> getRootClass() {
154        return rootClass;
155    }
156
157    public void setContextFactory(ContextFactory contextFactory) {
158        this.contextFactory = contextFactory;
159    }
160}
