043package org.dcm4che3.conf.core.olock;
044
045import org.dcm4che3.conf.core.api.Configuration;
046import org.dcm4che3.conf.core.util.ConfigNodeTraverser;
047import org.dcm4che3.util.Base64;
048import org.slf4j.Logger;
049import org.slf4j.LoggerFactory;
050
051import java.security.MessageDigest;
052import java.security.NoSuchAlgorithmException;
053import java.util.ArrayDeque;
054import java.util.Collection;
055import java.util.Deque;
056import java.util.Map;
057
058/**
059 * Calculates and sets olock hashes for nodes that have the _.hash property (ignores the previous value)
060 *
061 * @author Roman K
062 */
063public class OLockHashCalcFilter extends ConfigNodeTraverser.AConfigNodeFilter {
064
065    Logger log = LoggerFactory.getLogger(HashBasedOptimisticLockingConfiguration.class);
066
067    final Deque<byte[]> stack = new ArrayDeque<byte[]>();
068    private String ignoredKey;
069    private MessageDigest cript;
070
071    public OLockHashCalcFilter() {
072        stack.push(new byte[20]);
073
074    }
075
076    public OLockHashCalcFilter(String ignoredKey) {
077        this();
078        this.ignoredKey = ignoredKey;
079    }
080
081    /**
082     * Associative op
083     *
084     * @param one
085     * @param two
086     */
087    public void addHash(byte[] one, byte[] two) {
088
089        for (int i = 0; i < one.length; i++) {
090            one[i] = (byte) (one[i] + two[i]);
091        }
092    }
093
094    public byte[] getHash(String what) {
095        getCript().reset();
096        getCript().update(what.getBytes());
097        return getCript().digest();
098    }
099
100    public MessageDigest getCript() {
101
102        if (cript == null)
103            try {
104                cript = MessageDigest.getInstance("SHA-1");
105            } catch (NoSuchAlgorithmException e) {
106                throw new RuntimeException(e);
107            }
108
109        return cript;
110    }
111
112    public Deque<byte[]> getStack() {
113        return stack;
114    }
115
116
117    @Override
118    public void onPrimitiveNodeElement(Map<String, Object> containerNode, String key, Object value) {
119
120        // add up to hash
121        addHash(stack.peek(), getHash(String.valueOf(value)));
122
123    }
124
125    private String hashToString(byte[] hash) {
126        return Base64.toBase64(hash);
127    }
128
129    private boolean doIgnore(String key) {
130        return Configuration.OLOCK_HASH_KEY.equals(key) || ignoredKey != null && ignoredKey.equals(key);
131    }
132
133
134    @Override
135    public void beforeNode(Map<String, Object> node) {
136        stack.push(new byte[20]);
137    }
138
139    @Override
140    public void afterNode(Map<String, Object> node) {
141
142        byte[] pop = stack.pop();
143
144        // if this node is olock'd, save hash, otherwise addup to what will be consumed by the parent
145        if (node.containsKey(Configuration.OLOCK_HASH_KEY))
146            node.put(Configuration.OLOCK_HASH_KEY, hashToString(pop));
147        else
148            addHash(stack.peek(), getHash(hashToString(pop)));
149    }
150
151    @Override
152    public void beforeNodeElement(Map<String, Object> containerNode, String key, Object value) {
153        stack.push(new byte[20]);
154    }
155
156    @Override
157    public void afterNodeElement(Map<String, Object> containerNode, String key, Object value) {
158        byte[] valueHash = stack.pop();
159
160        // don't add for olock hashes and ignored keys
161        if (!doIgnore(key)) {
162//            String peekStr = hashToString(stack.peek());
163            addHash(stack.peek(), getHash(key + hashToString(valueHash)));
164//            log.info("Added hash of property '{}' ({} -> {})", key, peekStr, hashToString(stack.peek()));
165        }
166
167
168    }
169
170    @Override
171    public void beforeListElement(Collection list, int index, Object element) {
172        stack.push(new byte[20]);
173    }
174
175    @Override
176    public void afterListElement(Collection list, int index, Object element) {
177        byte[] listElementHash = stack.pop();
178        byte[] listHash = stack.pop();
179        stack.push(getHash(hashToString(listHash) + hashToString(listElementHash)));
180    }
181
182    @Override
183    public void onPrimitiveListElement(Collection list, Object element) {
184        addHash(stack.peek(), getHash(String.valueOf(element)));
185    }
186}
