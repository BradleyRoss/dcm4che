043package org.dcm4che3.conf.core.olock;
044
045import org.dcm4che3.conf.core.api.Configuration;
046import org.dcm4che3.conf.core.api.OptimisticLockException;
047import org.dcm4che3.conf.core.util.ConfigNodeTraverser.ADualNodeFilter;
048import org.dcm4che3.conf.core.Nodes;
049
050import java.util.*;
051
052/**
053 * Performs hash-based optimistic locking logic that is
054 * <ol>
055 *     <li>to detect conflicting changes an throw an exception if there are any</li>
056 *     <li>if there are none, make sure that all non-conflicting changes from both old and new nodes are preserved.</li>
057 * </ol>
058 *
059 * Merges old node and new node into a single resulting node.
060 * After running this filter, newNode becomes the resulting node.
061 * <br/>
062 * Changes the state of both nodes!
063 *
064 * @author Roman K
065 */
066class OLockMergeDualFilter extends ADualNodeFilter {
067
068    public static final String OLD_OLOCK_HASH_KEY = HashBasedOptimisticLockingConfiguration.OLD_OLOCK_HASH_KEY;
069
070    final Deque<Object> path = new ArrayDeque<Object>();
071
072    /**
073     * Is currently merging new (non-conflicting) stuff from backend into the new node (being persisted)
074     */
075    final Deque<Boolean> isMerging = new ArrayDeque<Boolean>();
076
077
078    public OLockMergeDualFilter() {
079        isMerging.push(false);
080    }
081
082    @Override
083    public void beforeNode(Map<String, Object> oldNode, Map<String, Object> newNode) {
084
085        // if this node is not olock-enabled, don't apply olock logic, just push current mode to stack once again
086        if (!newNode.containsKey(Configuration.OLOCK_HASH_KEY)) {
087            isMerging.push(isMerging.peek());
088            return;
089        }
090
091        if (!isMerging.peek()) {
092
093            // We are NOT merging now
094
095            if (newNode.get(OLD_OLOCK_HASH_KEY).equals(newNode.get(Configuration.OLOCK_HASH_KEY))) {
096                // If we met a olocked node where new node did not change, then we swap and turn on merging (from old to new)
097                isMerging.push(true);
098                swap(oldNode, newNode);
099            } else {
100                if (newNode.get(OLD_OLOCK_HASH_KEY).equals(oldNode.get(Configuration.OLOCK_HASH_KEY))) {
101                    // If we met a olocked node where new node changed, then check the hash in old node and if it's not changed - keep going
102                    isMerging.push(false);
103                } else {
104                    // If we met a olocked node where new node changed, and the hash in old node has also changed - that's an exception
105                    throw new OptimisticLockException(Nodes.toSimpleEscapedPath(path.descendingIterator()));
106                }
107            }
108        } else {
109
110            // We ARE merging now
111            // have to be careful since oldNode and newNode are swapped!
112            Map<String, Object> actualNewNode = oldNode;
113            Map<String, Object> actualOldNode = newNode;
114
115
116            if (actualNewNode.get(OLD_OLOCK_HASH_KEY).equals(actualNewNode.get(Configuration.OLOCK_HASH_KEY))) {
117                // If we met a olocked node where new node did not change, then keep on merging (from old to new)
118                isMerging.push(true);
119            } else {
120                if (actualNewNode.get(OLD_OLOCK_HASH_KEY).equals(actualOldNode.get(Configuration.OLOCK_HASH_KEY))) {
121                    // If we met a olocked node where new node changed, then check the hash in old node and if it's not changed, swap and switch to non-merging mode
122                    isMerging.push(false);
123                    swap(actualOldNode, actualNewNode);
124                } else {
125                    // If we met a olocked node where new node changed, and the hash in old node has also changed - that's an exception
126                    throw new OptimisticLockException(Nodes.toSimpleEscapedPath(path.descendingIterator()));
127                }
128            }
129        }
130    }
131
132    /**
133     * Swaps CONTENTS
134     *
135     * @param oldNode
136     * @param newNode
137     */
138    private void swap(Map<String, Object> oldNode, Map<String, Object> newNode) {
139        Map<String, Object> tmpNode = new LinkedHashMap<String, Object>();
140        tmpNode.putAll(oldNode);
141
142        oldNode.clear();
143        oldNode.putAll(newNode);
144
145        newNode.clear();
146        newNode.putAll(tmpNode);
147    }
148
149
150
151
152    @Override
153    public void afterNode(Map<String, Object> node1, Map<String, Object> node2) {
154        isMerging.pop();
155    }
156
157    @Override
158    public void beforeNodeProperty(String key) {
159        path.push(key);
160    }
161
162    @Override
163    public void afterNodeProperty(String key) {
164        path.pop();
165    }
166
167    @Override
168    public void beforeListElement(int index1, int index2) {
169        if (isMerging.peek())
170            path.push(Integer.toString(index1));
171        else
172            path.push(Integer.toString(index2));
173    }
174
175    @Override
176    public void afterListElement(int index1, int index2) {
177        path.pop();
178    }
179}
