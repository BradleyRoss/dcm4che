040package org.dcm4che3.conf.dicom;
041
042import org.dcm4che3.conf.api.TCGroupsProvider;
043import org.dcm4che3.conf.api.hl7.HL7Configuration;
044import org.dcm4che3.conf.core.api.Configuration;
045import org.dcm4che3.conf.core.api.ConfigurationException;
046import org.dcm4che3.conf.core.api.internal.ConfigTypeAdapter;
047import org.dcm4che3.net.Device;
048import org.dcm4che3.net.hl7.HL7Application;
049import org.dcm4che3.net.hl7.HL7DeviceExtension;
050
051import java.util.*;
052/**
053 * 
054 * <p>This class is part of the Maven module dcm4che-conf-dicom.</p>
055 *
056 */
057public class CommonDicomConfigurationWithHL7 extends CommonDicomConfiguration implements HL7Configuration {
058
059
060    public CommonDicomConfigurationWithHL7(Configuration configurationStorage, Map<Class, List<Class>> extensionsByClass) {
061        super(configurationStorage, extensionsByClass);
062    }
063    public CommonDicomConfigurationWithHL7(Configuration configurationStorage, Map<Class, List<Class>> extensionsByClass, boolean doCacheTCGroups) {
064        super(configurationStorage, extensionsByClass, doCacheTCGroups);
065    }
066
067
068    
069    public boolean registerHL7Application(String name) throws ConfigurationException {
070        return true;
071    }
072
073    
074    public void unregisterHL7Application(String name) throws ConfigurationException {
075    }
076
077    
078    public HL7Application findHL7Application(String name) throws ConfigurationException {
079        String pathForDeviceName = DicomPath.DeviceNameByHL7AppName.set("hl7AppName", name).path();
080
081        try {
082            Iterator search = lowLevelConfig.search(pathForDeviceName);
083            String deviceName = (String) search.next();
084
085            Device device = findDevice(deviceName);
086
087            return device.getDeviceExtension(HL7DeviceExtension.class).getHL7Application(name);
088        } catch (NoSuchElementException e) {
089            throw new ConfigurationException("HL7 app with name '" + name + "' not found", e);
090        } catch (Exception e) {
091            throw new ConfigurationException("Error while searching for HL7 app with name '" + name + "'", e);
092        }
093    }
094
095    
096    public String[] listRegisteredHL7ApplicationNames() throws ConfigurationException {
097        String hl7NamesPath = DicomPath.AllHL7AppNames.path();
098        List<String> list = new ArrayList<String>();
099        try {
100            Iterator search = lowLevelConfig.search(hl7NamesPath);
101            while (search.hasNext())
102                list.add((String) search.next());
103        } catch (Exception e) {
104            throw new ConfigurationException("Error while getting a list of HL7 app names", e);
105        }
106        return list.toArray(new String[list.size()]);
107    }
108
109    @SuppressWarnings("unchecked")
110    @Override
111    public <T> T getDicomConfigurationExtension(Class<T> clazz) {
112        // workaround for Weld, if we just return 'this' - it will replace it with a proxy
113        if (clazz.equals(HL7Configuration.class)) {
114            return (T) new HL7Configuration() {
115               
116                public boolean registerHL7Application(String name) throws ConfigurationException {
117                    return CommonDicomConfigurationWithHL7.this.registerHL7Application(name);
118                }
119
120                
121                public void unregisterHL7Application(String name) throws ConfigurationException {
122                    CommonDicomConfigurationWithHL7.this.unregisterHL7Application(name);
123                }
124
125                
126                public HL7Application findHL7Application(String name) throws ConfigurationException {
127                    return CommonDicomConfigurationWithHL7.this.findHL7Application(name);
128                }
129
130                
131                public String[] listRegisteredHL7ApplicationNames() throws ConfigurationException {
132                    return CommonDicomConfigurationWithHL7.this.listRegisteredHL7ApplicationNames();
133                }
134            };
135        }
136
137        return super.getDicomConfigurationExtension(clazz);
138    }
139}
