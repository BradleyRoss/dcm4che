040package org.dcm4che3.conf.dicom;
041
042import org.dcm4che3.conf.core.api.Path;
043import org.dcm4che3.conf.core.util.PathPattern;
044
045import java.util.HashMap;
046import java.util.Map;
047
048/**
049 * Single source for all the DicomConfiguration-related xpaths.
050 * 
051 * <p>This class is part of the Maven module dcm4che-conf-dicom.</p>
052 *
053 * @author Roman K
054 */
055public enum DicomPath {
056
057    DeviceNameByAEName,
058    DeviceNameByAENameAlias,
059    AllDeviceNames,
060    AllAETitles,
061    DeviceNameByHL7AppName,
062    AllHL7AppNames,
063    ConfigRoot,
064    TCGroups,
065    AllTCsOfAllAEsWithTCGroupExt,
066    DeviceNameByAEUUID,
067    DeviceNameByUUID,
068    DeviceUUIDByAnyUUID,
069    DeviceByNameForWrite,
070    DeviceByNameForRead,
071    AETransferCapabilities;
072
073    public static final Path TC_GROUPS_PATH = new Path("dicomConfigurationRoot","globalConfiguration","dcmTransferCapabilities");
074    public static final Path CONFIG_ROOT_PATH = new Path("dicomConfigurationRoot");
075
076
077    public static final Map<DicomPath, String> PATHS = new HashMap<DicomPath, String>();
078    public static final Map<DicomPath, PathPattern> PATH_PATTERNS = new HashMap<DicomPath, PathPattern>();
079
080    static {
081        // search
082        PATHS.put(/**************/AllAETitles, "/dicomConfigurationRoot/dicomDevicesRoot/*/dicomNetworkAE/*/dicomAETitle | /dicomConfigurationRoot/dicomDevicesRoot/*/dicomNetworkAE/*/dcmAETitleAliases");
083
084        PATHS.put(/*******/DeviceNameByAEName, "/dicomConfigurationRoot/dicomDevicesRoot/*[dicomNetworkAE[@name='{aeName}']]/dicomDeviceName");
085        PATHS.put(/**/DeviceNameByAENameAlias, "/dicomConfigurationRoot/dicomDevicesRoot/*[dicomNetworkAE[*/dcmAETitleAliases='{aeNameAlias}']]/dicomDeviceName");
086
087        PATHS.put(/***********/AllDeviceNames, "/dicomConfigurationRoot/dicomDevicesRoot/*/dicomDeviceName");
088        PATHS.put(/***********/AllHL7AppNames, "/dicomConfigurationRoot/dicomDevicesRoot/*/deviceExtensions/HL7DeviceExtension/hl7Apps/*/hl7ApplicationName");
089        PATHS.put(/***/DeviceNameByHL7AppName, "/dicomConfigurationRoot/dicomDevicesRoot/*[deviceExtensions/HL7DeviceExtension/hl7Apps/*[hl7ApplicationName='{hl7AppName}']]/dicomDeviceName");
090        PATHS.put(/*******/DeviceNameByAEUUID, "/dicomConfigurationRoot/dicomDevicesRoot/*[dicomNetworkAE/*[_.uuid='{aeUUID}']]/dicomDeviceName");
091        PATHS.put(/*********/DeviceNameByUUID, "/dicomConfigurationRoot/dicomDevicesRoot/*[_.uuid='{deviceUUID}']/dicomDeviceName");
092
093        // Get device UUID of a device that contains (or itself is) an object with the specified UUID
094        PATHS.put(/******/DeviceUUIDByAnyUUID, "/dicomConfigurationRoot/dicomDevicesRoot/*[.//_.uuid='{UUID}']/_.uuid");
095
096        // single-result getNode (also can be used to store nodes)
097        PATHS.put(/***************/ConfigRoot, "/dicomConfigurationRoot");
098        PATHS.put(/*****/DeviceByNameForWrite, "/dicomConfigurationRoot/dicomDevicesRoot[@name='{deviceName}']");
099
100        PATHS.put(/******/DeviceByNameForRead, "/dicomConfigurationRoot/dicomDevicesRoot/{deviceName}");
101
102        // Transfer capabilities
103        PATHS.put(/*****************/TCGroups, TC_GROUPS_PATH.toSimpleEscapedXPath());
104        PATHS.put(AllTCsOfAllAEsWithTCGroupExt, "dicomNetworkAE/*[aeExtensions/TCGroupConfigAEExtension]/dcmTransferCapability");
105        PATHS.put(/****/AETransferCapabilities, "/dcmTransferCapability");
106
107
108        for (Map.Entry<DicomPath, String> entry : PATHS.entrySet()) {
109            PATH_PATTERNS.put(entry.getKey(), new PathPattern(entry.getValue()));
110        }
111    }
112
113    public static Path devicePath(String name) {
114        return new Path("dicomConfigurationRoot", "dicomDevicesRoot", name);
115    }
116
117    public static void validateDevicePath(Path path) {
118        if (path.getPathItems().size()>3 ||
119                !path.getPathItems().get(0).equals("dicomConfigurationRoot") ||
120                !path.getPathItems().get(1).equals("dicomDevicesRoot")) {
121            throw new IllegalArgumentException("Unexpected path:" + path);
122        }
123    }
124
125    /**
126     * Starts chaining
127     *
128     * @param paramName
129     * @param value
130     * @return
131     */
132    public PathPattern.PathCreator set(String paramName, String value) {
133        return PATH_PATTERNS.get(this).set(paramName, value);
134    }
135
136
137    public PathPattern.PathParser parse(String path) {
138        return PATH_PATTERNS.get(this).parse(path);
139    }
140
141    public PathPattern.PathParser parseIfMatches(String path) {
142        return PATH_PATTERNS.get(this).parseIfMatches(path);
143    }
144
145    /**
146     * Gets path as is
147     *
148     * @return
149     */
150    public String path() {
151        return PATHS.get(this);
152    }
153
154}
