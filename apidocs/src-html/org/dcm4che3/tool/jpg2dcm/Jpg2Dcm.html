159    public void convert(CommandLine cl, File jpgFile, File dcmFile)
160            throws IOException {
161        jpgHeaderLen = 0;
162        jpgLen = (int) jpgFile.length();
163        DataInputStream jpgInput = new DataInputStream(new BufferedInputStream(
164                new FileInputStream(jpgFile)));
165        try {
166            Attributes attrs = new Attributes();
167            attrs.setString(Tag.SOPClassUID, VR.UI, UID.SecondaryCaptureImageStorage);
168            try {
169                if (cl.hasOption("mpeg") && cl.hasOption("c"))
170                    attrs = SAXReader.parse(cl.getOptionValue("c"));
171                else if (cl.hasOption("c"))
172                    attrs = SAXReader.parse(cl.getOptionValue("c"));
173            } catch (Exception e) {
174                throw new FileNotFoundException(
175                        "Configuration XML file not found");
176            }
177            attrs.setString(Tag.SpecificCharacterSet, VR.CS, charset);
178            if (noAPPn || missingRowsColumnsSamplesPMI(attrs)) {
179                readHeader(attrs, jpgInput);
180            }
181            ensureUS(attrs, Tag.BitsAllocated, 8);
182            ensureUS(attrs, Tag.BitsStored, attrs.getInt(Tag.BitsAllocated,
183                    (buffer[jpgHeaderLen] & 0xff) > 8 ? 16 : 8));
184            ensureUS(
185                    attrs,
186                    Tag.HighBit,
187                    attrs.getInt(Tag.BitsStored, (buffer[jpgHeaderLen] & 0xff)) - 1);
188            ensureUS(attrs, Tag.PixelRepresentation, 0);
189            ensureUID(attrs, Tag.StudyInstanceUID);
190            ensureUID(attrs, Tag.SeriesInstanceUID);
191            ensureUID(attrs, Tag.SOPInstanceUID);
192            Date now = new Date();
193            attrs.setDate(Tag.InstanceCreationDate, VR.DA, now);
194            attrs.setDate(Tag.InstanceCreationTime, VR.TM, now);
195            Attributes fmi = attrs.createFileMetaInformation(transferSyntax);
196            DicomOutputStream dos = new DicomOutputStream(dcmFile);
197            try {
198                dos.writeDataset(fmi, attrs);
199                dos.writeHeader(Tag.PixelData, VR.OB, -1);
200                if (!cl.hasOption("mpeg")) {
201                    dos.writeHeader(Tag.Item, null, 0);
202                    dos.writeHeader(Tag.Item, null, (jpgLen + 1) & ~1);
203                    dos.write(buffer, 0, jpgHeaderLen);
204                }
205                int r;
206                while ((r = jpgInput.read(buffer)) > 0) {
207                    dos.write(buffer, 0, r);
208                }
209                if (!cl.hasOption("mpeg")) {
210                    if ((jpgLen & 1) != 0) {
211                        dos.write(0);
212                    }
213                }
214                dos.writeHeader(Tag.SequenceDelimitationItem, null, 0);
215            } finally {
216                dos.close();
217            }
218        } finally {
219            jpgInput.close();
220        }
221    }

223    private boolean missingRowsColumnsSamplesPMI(Attributes attrs) {
224        return !(attrs.containsValue(Tag.Rows)
225                && attrs.containsValue(Tag.Columns)
226                && attrs.containsValue(Tag.SamplesPerPixel) && attrs
227                    .containsValue(Tag.PhotometricInterpretation));
228    }

230    private void readHeader(Attributes attrs, DataInputStream jpgInput)
231            throws IOException {
232        if (jpgInput.read() != FF || jpgInput.read() != SOI
233                || jpgInput.read() != FF) {
234            throw new IOException("JPEG stream does not start with FF D8 FF");
235        }
236        int marker = jpgInput.read();
237        int segmLen;
238        boolean seenSOF = false;
239        buffer[0] = (byte) FF;
240        buffer[1] = (byte) SOI;
241        buffer[2] = (byte) FF;
242        buffer[3] = (byte) marker;
243        jpgHeaderLen = 4;
244        while (marker != SOS) {
245            segmLen = jpgInput.readUnsignedShort();
246            if (buffer.length < jpgHeaderLen + segmLen + 2) {
247                growBuffer(jpgHeaderLen + segmLen + 2);
248            }
249            buffer[jpgHeaderLen++] = (byte) (segmLen >>> 8);
250            buffer[jpgHeaderLen++] = (byte) segmLen;
251            jpgInput.readFully(buffer, jpgHeaderLen, segmLen - 2);
252            if ((marker & 0xf0) == SOF && marker != DHT && marker != DAC) {
253                seenSOF = true;
254                int p = buffer[jpgHeaderLen] & 0xff;
255                int y = ((buffer[jpgHeaderLen + 1] & 0xff) << 8)
256                        | (buffer[jpgHeaderLen + 2] & 0xff);
257                int x = ((buffer[jpgHeaderLen + 3] & 0xff) << 8)
258                        | (buffer[jpgHeaderLen + 4] & 0xff);
259                int nf = buffer[jpgHeaderLen + 5] & 0xff;
260                attrs.setInt(Tag.SamplesPerPixel, VR.US, nf);
261                if (nf == 3) {
262                    attrs.setString(Tag.PhotometricInterpretation, VR.CS,
263                            "YBR_FULL_422");
264                    attrs.setInt(Tag.PlanarConfiguration, VR.US, 0);
265                } else {
266                    attrs.setString(Tag.PhotometricInterpretation, VR.CS,
267                            "MONOCHROME2");
268                }
269                attrs.setInt(Tag.Rows, VR.US, y);
270                attrs.setInt(Tag.Columns, VR.US, x);
271                attrs.setInt(Tag.BitsAllocated, VR.US, p > 8 ? 16 : 8);
272                attrs.setInt(Tag.BitsStored, VR.US, p);
273                attrs.setInt(Tag.HighBit, VR.US, p - 1);
274                attrs.setInt(Tag.PixelRepresentation, VR.US, 0);
275            }
276            if (noAPPn & (marker & 0xf0) == APP) {
277                jpgLen -= segmLen + 2;
278                jpgHeaderLen -= 4;
279            } else {
280                jpgHeaderLen += segmLen - 2;
281            }
282            if (jpgInput.read() != FF) {
283                throw new IOException("Missing SOS segment in JPEG stream");
284            }
285            marker = jpgInput.read();
286            buffer[jpgHeaderLen++] = (byte) FF;
287            buffer[jpgHeaderLen++] = (byte) marker;
288        }
289        if (!seenSOF) {
290            throw new IOException("Missing SOF segment in JPEG stream");
291        }
292    }

294    private void growBuffer(int minSize) {
295        int newSize = buffer.length << 1;
296        while (newSize < minSize) {
297            newSize <<= 1;
298        }
299        byte[] tmp = new byte[newSize];
300        System.arraycopy(buffer, 0, tmp, 0, jpgHeaderLen);
301        buffer = tmp;
302    }

304    private void ensureUID(Attributes attrs, int tag) {
305        if (!attrs.containsValue(tag)) {
306            attrs.setString(tag, VR.UI, UIDUtils.createUID());
307        }
308    }
309
310    private void ensureUS(Attributes attrs, int tag, int val) {
311        if (!attrs.containsValue(tag)) {
312            attrs.setInt(tag, VR.US, val);
313        }
314    }

316    public static void main(String[] args) {
317        try {
318            CommandLine cl = parse(args);
319            Jpg2Dcm jpg2Dcm = new Jpg2Dcm();
320            if (cl.hasOption(LONG_OPT_CHARSET)) {
321                jpg2Dcm.setCharset(cl.getOptionValue(LONG_OPT_CHARSET));
322            }
323            if (cl.hasOption(LONG_OPT_UID_PREFIX)) {
324                UIDUtils.setRoot(cl.getOptionValue(LONG_OPT_UID_PREFIX));
325            }
326            if (cl.hasOption(LONG_OPT_MPEG)) {
327                jpg2Dcm.setTransferSyntax(UID.MPEG2);
328            }
329            if (cl.hasOption(LONG_OPT_TRANSFER_SYNTAX)) {
330                jpg2Dcm.setTransferSyntax(cl
331                        .getOptionValue(LONG_OPT_TRANSFER_SYNTAX));
332            }
333            jpg2Dcm.setNoAPPn(cl.hasOption(LONG_OPT_NO_APPN));
334
335            @SuppressWarnings("rawtypes")
336            List argList = cl.getArgList();
337            File jpgFile = new File((String) argList.get(0));
338            File dcmFile = new File((String) argList.get(1));
339            long start = System.currentTimeMillis();
340            jpg2Dcm.convert(cl, jpgFile, dcmFile);
341            long fin = System.currentTimeMillis();
342            LOG.info("Encapsulated " + jpgFile + " to " + dcmFile + " in "
343                    + (fin - start) + "ms.");
344        } catch (IOException e) {
345            e.printStackTrace();
346        }
347    }

349    private static CommandLine parse(String[] args) {
350        Options opts = new Options();
351
352        OptionBuilder.withArgName("code");
353        OptionBuilder.hasArg();
354        OptionBuilder.withDescription(OPT_CHARSET_DESC);
355        OptionBuilder.withLongOpt(LONG_OPT_CHARSET);
356        opts.addOption(OptionBuilder.create());
357
358        OptionBuilder.withArgName("file");
359        OptionBuilder.hasArg();
360        OptionBuilder.withDescription(OPT_AUGMENT_CONFIG_DESC);
361        opts.addOption(OptionBuilder.create("c"));
362
363        OptionBuilder.withArgName("file");
364        OptionBuilder.hasArg();
365        OptionBuilder.withDescription(OPT_REPLACE_CONFIG_DESC);
366        opts.addOption(OptionBuilder.create("C"));
367
368        OptionBuilder.withArgName("prefix");
369        OptionBuilder.hasArg();
370        OptionBuilder.withDescription(OPT_UID_PREFIX_DESC);
371        OptionBuilder.withLongOpt(LONG_OPT_UID_PREFIX);
372        opts.addOption(OptionBuilder.create());
373
374        OptionBuilder.withArgName("uid");
375        OptionBuilder.hasArg();
376        OptionBuilder.withDescription(OPT_TRANSFER_SYNTAX_DESC);
377        OptionBuilder.withLongOpt(LONG_OPT_TRANSFER_SYNTAX);
378        opts.addOption(OptionBuilder.create());
379
380        opts.addOption(null, LONG_OPT_MPEG, false, OPT_MPEG_DESC);
381
382        opts.addOption(null, LONG_OPT_NO_APPN, false, OPT_NO_APPN_DESC);
383
384        opts.addOption("h", "help", false, OPT_HELP_DESC);
385        opts.addOption("V", "version", false, OPT_VERSION_DESC);
386
387        CommandLine cl = null;
388        try {
389            cl = new PosixParser().parse(opts, args);
390        } catch (ParseException e) {
391            exit("jpg2dcm: " + e.getMessage());
392            throw new RuntimeException("unreachable");
393        }
394        if (cl.hasOption('V')) {
395            Package p = Jpg2Dcm.class.getPackage();
396            LOG.info("jpg2dcm v" + p.getImplementationVersion());
397            System.exit(0);
398        }
399        if (cl.hasOption('h') || cl.getArgList().size() != 2) {
400            HelpFormatter formatter = new HelpFormatter();
401            formatter.printHelp(USAGE, DESCRIPTION, opts, EXAMPLE);
402            System.exit(0);
403        }
404
405        return cl;
406    }

408    private static void exit(String msg) {
409        LOG.error(msg);
410        LOG.error("Try 'jpg2dcm -h' for more information.");
411        System.exit(1);
412    }
