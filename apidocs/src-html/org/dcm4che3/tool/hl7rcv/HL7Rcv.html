098    private final Device device = new Device("hl7rcv");
099    private final HL7DeviceExtension hl7Ext = new HL7DeviceExtension();
100    private final HL7Application hl7App = new HL7Application("*");

103    private String charset;

106    private final HL7MessageListener handler = new HL7MessageListener() {
107
108        public byte[] onMessage(HL7Application hl7App, Connection conn,
109                Socket s, HL7Segment msh, byte[] msg, int off, int len,
110                int mshlen) throws HL7Exception {
111            try {
112                return HL7Rcv.this.onMessage(msh, msg, off, len);
113            } catch (Exception e) {
114                throw new HL7Exception(HL7Exception.AE, e);
115            }
116        }
117    };

119    public HL7Rcv() throws IOException {
120        conn.setProtocol(Protocol.HL7);
121        device.addDeviceExtension(hl7Ext);
122        device.addConnection(conn);
123        hl7Ext.addHL7Application(hl7App);
124        hl7App.setAcceptedMessageTypes("*");
125        hl7App.addConnection(conn);
126        hl7App.setHL7MessageListener(handler);
127    }

146    private static CommandLine parseComandLine(String[] args)
147            throws ParseException {
148        Options opts = new Options();
149        addOptions(opts);
150        CLIUtils.addSocketOptions(opts);
151        CLIUtils.addTLSOptions(opts);
152        CLIUtils.addCommonOptions(opts);
153        return CLIUtils.parseComandLine(args, opts, rb, HL7Rcv.class);
154    }
155
156    @SuppressWarnings("static-access")
157    public static void addOptions(Options opts) {
158        opts.addOption(null, "ignore", false, rb.getString("ignore"));
159        opts.addOption(OptionBuilder
160                .hasArg()
161                .withArgName("path")
162                .withDescription(rb.getString("directory"))
163                .withLongOpt("directory")
164                .create(null));
165        opts.addOption(OptionBuilder
166                .withLongOpt("xsl")
167                .hasArg()
168                .withArgName("xsl-file")
169                .withDescription(rb.getString("xsl"))
170                .create("x"));
171        opts.addOption(OptionBuilder
172                .withLongOpt("xsl-param")
173                .hasArgs()
174                .withValueSeparator('=')
175                .withArgName("name=value")
176                .withDescription(rb.getString("xsl-param"))
177                .create(null));
178        opts.addOption(OptionBuilder
179                .withLongOpt("charset")
180                .hasArg()
181                .withArgName("name")
182                .withDescription(rb.getString("charset"))
183                .create(null));
184        opts.addOption(OptionBuilder
185                .hasArg()
186                .withArgName("[ip:]port")
187                .withDescription(rb.getString("bind-server"))
188                .withLongOpt("bind")
189                .create("b"));
190        opts.addOption(OptionBuilder
191                .hasArg()
192                .withArgName("ms")
193                .withDescription(rb.getString("idle-timeout"))
194                .withLongOpt("idle-timeout")
195                .create(null));
196    }

198    public static void main(String[] args) {
199        try {
200            CommandLine cl = parseComandLine(args);
201            HL7Rcv main = new HL7Rcv();
202            configure(main, cl);
203            ExecutorService executorService = Executors.newCachedThreadPool();
204            ScheduledExecutorService scheduledExecutorService = 
205                    Executors.newSingleThreadScheduledExecutor();
206            main.device.setScheduledExecutor(scheduledExecutorService);
207            main.device.setExecutor(executorService);
208            main.device.bindConnections();
209        } catch (ParseException e) {
210            System.err.println("hl7rcv: " + e.getMessage());
211            System.err.println(rb.getString("try"));
212            System.exit(2);
213        } catch (Exception e) {
214            System.err.println("hl7rcv: " + e.getMessage());
215            e.printStackTrace();
216            System.exit(2);
217        }
218    }

220    private static void configure(HL7Rcv main, CommandLine cl)
221            throws Exception, MalformedURLException, ParseException,
222            IOException {
223        if (!cl.hasOption("ignore"))
224            main.setStorageDirectory(
225                    cl.getOptionValue("directory", "."));
226        if (cl.hasOption("x")) {
227            String s = cl.getOptionValue("x");
228            main.setXSLT(new File(s).toURI().toURL());
229            main.setXSLTParameters(cl.getOptionValues("xsl-param"));
230        }
231        main.setCharacterSet(cl.getOptionValue("charset"));
232        configureBindServer(main.conn, cl);
233        CLIUtils.configure(main.conn, cl);
234    }
235
236    private static void configureBindServer(Connection conn, CommandLine cl)
237            throws ParseException {
238        if (!cl.hasOption("b"))
239            throw new MissingOptionException(
240                    CLIUtils.rb.getString("missing-bind-opt"));
241        String aeAtHostPort = cl.getOptionValue("b");
242        String[] hostAndPort = StringUtils.split(aeAtHostPort, ':');
243        int portIndex = hostAndPort.length - 1;
244        conn.setPort(Integer.parseInt(hostAndPort[portIndex]));
245        if (portIndex > 0)
246            conn.setHostname(hostAndPort[0]);
247    }
248
249    private byte[] onMessage(HL7Segment msh, byte[] msg, int off, int len)
250                throws Exception {
251            if (storageDir != null)
252                storeToFile(msg, off, len, 
253                        new File(
254                            new File(storageDir, msh.getMessageType()),
255                            msh.getField(9, "_NULL_")));
256            return (tpls == null)
257                ? HL7Message.makeACK(msh, HL7Exception.AA, null).getBytes(null)
258                : xslt(msh, msg, off, len);
259    }
260
261    private void storeToFile(byte[] msg, int off, int len, File f)
262            throws FileNotFoundException, IOException {
263        Connection.LOG.info("M-WRITE {}", f);
264        f.getParentFile().mkdirs();
265        FileOutputStream out = new FileOutputStream(f);
266        try {
267            out.write(msg, off, len);
268        } finally {
269            out.close();
270        }
271    }
272
273    private byte[] xslt(HL7Segment msh, byte[] msg, int off, int len)
274            throws Exception {
275        String charsetName = HL7Charset.toCharsetName(msh.getField(17, charset));
276        ByteArrayOutputStream out = new ByteArrayOutputStream();
277        TransformerHandler th = factory.newTransformerHandler(tpls);
278        Transformer t = th.getTransformer();
279        t.setParameter("MessageControlID", HL7Segment.nextMessageControlID());
280        t.setParameter("DateTimeOfMessage", HL7Segment.timeStamp(new Date()));
281        if (xsltParams != null)
282            for (int i = 1; i < xsltParams.length; i++, i++)
283                t.setParameter(xsltParams[i-1], xsltParams[i]);
284        th.setResult(new SAXResult(new HL7ContentHandler(
285                new OutputStreamWriter(out, charsetName))));
286        new HL7Parser(th).parse(new InputStreamReader(
287                new ByteArrayInputStream(msg, off, len),
288                charsetName));
289        return out.toByteArray();
290    }
