039package org.dcm4che3.tool.syslogd;
040
041import java.io.File;
042import java.io.FileOutputStream;
043import java.io.IOException;
044import java.net.InetAddress;
045import java.net.MalformedURLException;
046import java.util.ResourceBundle;
047import java.util.concurrent.ExecutorService;
048import java.util.concurrent.Executors;
049import java.util.concurrent.ScheduledExecutorService;
050
051import org.apache.commons.cli.CommandLine;
052import org.apache.commons.cli.MissingOptionException;
053import org.apache.commons.cli.OptionBuilder;
054import org.apache.commons.cli.Options;
055import org.apache.commons.cli.ParseException;
056import org.dcm4che3.net.Connection;
057import org.dcm4che3.net.Connection.Protocol;
058import org.dcm4che3.net.Device;
059import org.dcm4che3.net.audit.AuditRecordHandler;
060import org.dcm4che3.net.audit.AuditRecordRepository;
061import org.dcm4che3.tool.common.CLIUtils;
062import org.dcm4che3.util.StringUtils;
063
064/**
065 * @author Gunter Zeilinger <gunterze@gmail.com>
066 *
067 */
068public class Syslogd {
069
070    private static final ResourceBundle rb =
071            ResourceBundle.getBundle("org.dcm4che3.tool.syslogd.messages");
072
073    private final Device device = new Device("syslogd");
074    private final AuditRecordRepository arr = new AuditRecordRepository();
075    private final Connection conn = new Connection();
076    private File storageDir;
077    private final AuditRecordHandler handler = new AuditRecordHandler() {
078
079        public void onMessage(byte[] data, int xmlOffset, int xmlLength,
080                Connection conn, InetAddress from) {
081            Syslogd.this.onMessage(data, xmlOffset, xmlLength, conn, from);
082        }
083    };
084
085    public Syslogd() {
086        device.addDeviceExtension(arr);
087        device.addConnection(conn);
088        arr.setAuditRecordHandler(handler);
089    }
090
091    public void setStorageDirectory(String storageDir) {
092        this.storageDir = storageDir != null ? new File(storageDir) : null;
093    }
094
095    private static CommandLine parseComandLine(String[] args)
096            throws ParseException {
097        Options opts = new Options();
098        addOptions(opts);
099        CLIUtils.addSocketOptions(opts);
100        CLIUtils.addTLSOptions(opts);
101        CLIUtils.addCommonOptions(opts);
102        return CLIUtils.parseComandLine(args, opts, rb, Syslogd.class);
103    }
104
105    @SuppressWarnings("static-access")
106    public static void addOptions(Options opts) {
107        opts.addOption(null, "ignore", false, rb.getString("ignore"));
108        opts.addOption(OptionBuilder
109                .hasArg()
110                .withArgName("path")
111                .withDescription(rb.getString("directory"))
112                .withLongOpt("directory")
113                .create(null));
114        opts.addOption(OptionBuilder
115                .hasArg()
116                .withArgName("[ip:]port")
117                .withDescription(rb.getString("bind-server"))
118                .withLongOpt("bind")
119                .create("b"));
120        opts.addOption(OptionBuilder
121                .hasArg()
122                .withArgName("ms")
123                .withDescription(rb.getString("idle-timeout"))
124                .withLongOpt("idle-timeout")
125                .create(null));
126        opts.addOption(null, "udp", false, rb.getString("udp"));
127    }
128
129    public static void main(String[] args) {
130        try {
131            CommandLine cl = parseComandLine(args);
132            Syslogd main = new Syslogd();
133            configure(main, cl);
134            ExecutorService executorService = Executors.newCachedThreadPool();
135            ScheduledExecutorService scheduledExecutorService = 
136                    Executors.newSingleThreadScheduledExecutor();
137            main.device.setScheduledExecutor(scheduledExecutorService);
138            main.device.setExecutor(executorService);
139            main.device.bindConnections();
140        } catch (ParseException e) {
141            System.err.println("syslogd: " + e.getMessage());
142            System.err.println(rb.getString("try"));
143            System.exit(2);
144        } catch (Exception e) {
145            System.err.println("syslogd: " + e.getMessage());
146            e.printStackTrace();
147            System.exit(2);
148        }
149    }
150
151    private static void configure(Syslogd main, CommandLine cl)
152            throws Exception, MalformedURLException, ParseException,
153            IOException {
154        if (!cl.hasOption("ignore"))
155            main.setStorageDirectory(
156                    cl.getOptionValue("directory", "."));
157        configureBindServer(main.conn, cl);
158        main.setProtocol(toProtocol(cl));
159        main.arr.addConnection(main.conn);
160        CLIUtils.configure(main.conn, cl);
161    }
162
163    private void setProtocol(Protocol protocol) {
164        conn.setProtocol(protocol);
165        arr.addConnection(conn);
166    }
167
168    private static void configureBindServer(Connection conn, CommandLine cl)
169            throws ParseException {
170        if (!cl.hasOption("b"))
171            throw new MissingOptionException(
172                    CLIUtils.rb.getString("missing-bind-opt"));
173        String aeAtHostPort = cl.getOptionValue("b");
174        String[] hostAndPort = StringUtils.split(aeAtHostPort, ':');
175        int portIndex = hostAndPort.length - 1;
176        conn.setPort(Integer.parseInt(hostAndPort[portIndex]));
177        if (portIndex > 0)
178            conn.setHostname(hostAndPort[0]);
179    }
180
181    private static Protocol toProtocol(CommandLine cl) {
182        return cl.hasOption("udp")
183                ? Connection.Protocol.SYSLOG_UDP
184                : Connection.Protocol.SYSLOG_TLS;
185    }
186
187    private void onMessage(byte[] data, int off, int len, Connection conn,
188            InetAddress from) {
189            if (storageDir != null) 
190                try {
191                    storeToFile(data, off, len,
192                            File.createTempFile("syslog", ".xml", storageDir));
193                } catch (IOException e) {
194                    Connection.LOG.warn("Failed to store received message", e);
195                }
196    }
197
198    private void storeToFile(byte[] data, int off, int len, File f)
199            throws IOException {
200        Connection.LOG.info("M-WRITE {}", f);
201        f.getParentFile().mkdirs();
202        FileOutputStream out = new FileOutputStream(f);
203        try {
204            out.write(data, off, len);
205        } finally {
206            out.close();
207        }
208    }
