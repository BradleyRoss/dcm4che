038package org.dcm4che3.tool.wadors.test;
039
040import java.io.File;
041import java.io.IOException;
042import java.util.ArrayList;
043
044import org.dcm4che3.tool.common.test.TestResult;
045import org.dcm4che3.tool.common.test.TestTool;
046import org.dcm4che3.tool.wadors.WadoRS;
047
048/**
049 * WADO-RS client tool for tests.
050 * 
051 * @author Hesham Elbadawi <bsdreko@gmail.com>
052 */
053public class WadoRSTool implements TestTool {
054
055    private ArrayList<String> mediaTypesWithTS = new ArrayList<String>();
056    private final String url;
057    private final File retrieveDir;
058    private WadoRSResult result;
059
060    public WadoRSTool(String url, File retrieveDir) {
061        this.url = url;
062        this.retrieveDir = retrieveDir;
063    }
064
065    public void wadoRS(String testDescription, boolean dumpHeader) throws IOException {
066        long t1, t2;
067        WadoRS wadors = new WadoRS(getUrl(),getRetrieveDir());
068        t1 = System.currentTimeMillis();
069        wadors.setAcceptType(getMediaTypesWithTS().toArray(new String[getMediaTypesWithTS().size()]));
070        wadors.setDumpHeaders(dumpHeader);
071        wadors.wadors();
072        t2 = System.currentTimeMillis();
073        init(new WadoRSResult(testDescription, t2-t1, wadors.getResponse()));
074    }
075
076    /**
077     * Add accepted media type and (optional) transfer syntax UID.
078     * 
079     * @param mediaType
080     *            media type (e.g. "application/dicom", "application/dicom+xml",
081     *            "application/json", "application/octet-stream",
082     *            "image/dicom+jpeg", ...)
083     * @param transferSyntax
084     *            optional transfer syntax UID to accept (please note that
085     *            specifying a transfer syntax UID is not allowed for all media
086     *            types)
087     */
088    public void addAcceptType(String mediaType, String transferSyntax) {
089        mediaTypesWithTS.add(mediaType + (transferSyntax != null ? ";" + transferSyntax : ""));
090    }
091
092    public ArrayList<String> getMediaTypesWithTS() {
093        return this.mediaTypesWithTS;
094    }
095
096    public void setMediaTypesWithTS(ArrayList<String> mediaTypesWithTS) {
097        this.mediaTypesWithTS = mediaTypesWithTS;
098    }
099
100    public String getUrl() {
101        return this.url;
102    }
103
104    public File getRetrieveDir() {
105        return this.retrieveDir;
106    }
107
108    public void init(TestResult resultIn) {
109        this.result = (WadoRSResult) resultIn;
110    }
111
112    public WadoRSResult getResult() {
113        return this.result;
114    }
115
116}
